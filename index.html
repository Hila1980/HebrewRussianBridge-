function openTab(tabName) { // מסתיר את כל התוכן var tabcontent = document.getElementsByClassName("tab-content"); for (var i = 0; i < tabcontent.length; i++) { tabcontent[i].style.display = "none"; tabcontent[i].classList.remove("active"); } // מסיר את המחלקה 'active' מכל הלשוניות var tablinks = document.getElementsByClassName("tab"); for (var i = 0; i < tablinks.length; i++) { tablinks[i].classList.remove("active"); } // מציג את התוכן הנבחר ומוסיף את המחלקה 'active' ללשונית המתאימה document.getElementById(tabName).style.display = "block"; document.getElementById(tabName).classList.add("active"); // מוצא את הלשונית הנכונה ומסמן אותה כפעילה for (var i = 0; i < tablinks.length; i++) { if (tablinks[i].textContent.includes(getTabName(tabName))) { tablinks[i].classList.add("active"); break; } } }
